วันที่และเวลาภาษาไทย

การแปลงวันที่และเวลาเป็นภาษาไทยด้วย JavaScript

ตัวอย่างรูปแบบวันที่และเวลาภาษาไทย

รูปแบบที่ 1: วันที่แบบสั้น

toLocaleDateString("th-TH", {day: "numeric", month: "short", year: "numeric", weekday: "short"})

รูปแบบที่ 2: วันที่แบบยาว

toLocaleDateString("th-TH", {day: "numeric", month: "long", year: "numeric", weekday: "long"})

รูปแบบที่ 3: เวลาพร้อมวันที่แบบสั้น

toLocaleTimeString("th-TH", {day: "numeric", month: "short", year: "numeric", weekday: "short"})

รูปแบบที่ 4: วันที่และเวลาแบบเต็ม

toLocaleString('th-TH', {dateStyle: 'full', timeStyle: 'medium'})

รูปแบบที่ 5: วันที่แบบเต็ม

toLocaleDateString('th-TH', {dateStyle: 'full'})

รูปแบบที่ 6: เวลาแบบกลาง

toLocaleTimeString('th-TH', {timeStyle: 'medium'})

รูปแบบที่ 7: วันที่และเวลาแบบสั้น

toLocaleString('th-TH', {dateStyle: 'short', timeStyle: 'short'})

รูปแบบที่ 8: วันที่แบบสั้นและเวลาแบบเต็ม

วันที่แบบสั้น + เวลาแบบเต็ม (ชั่วโมง:นาที:วินาที น.)

รูปแบบที่ 9: วันที่แบบสั้นและเวลาแบบสั้น

วันที่แบบสั้น + เวลาแบบสั้น (ชั่วโมง:นาที น.)

เวลาปัจจุบัน (อัพเดททุกวินาที)

โค้ด JavaScript สำหรับแปลงวันที่และเวลาภาษาไทย 

// รูปแบบที่ 1: วันที่แบบสั้น
var convertDate11 = new Date().toLocaleDateString("th-TH", {
  day: "numeric",
  month: "short",
  year: "numeric",
  weekday: "short"
});

// รูปแบบที่ 2: วันที่แบบยาว
var convertDate12 = new Date().toLocaleDateString("th-TH", {
  day: "numeric",
  month: "long",
  year: "numeric",
  weekday: "long"
});

// รูปแบบที่ 3: เวลาพร้อมวันที่แบบสั้น
var convertDate13 = new Date().toLocaleTimeString("th-TH", {
  day: "numeric",
  month: "short",
  year: "numeric",
  weekday: "short"
});

// รูปแบบที่ 4: วันที่และเวลาแบบเต็ม
const now = new Date();
const convertDate14 = now.toLocaleString('th-TH', {
  dateStyle: 'full',
  timeStyle: 'medium'
});

// รูปแบบที่ 5: วันที่แบบเต็ม
const convertDate15 = now.toLocaleDateString('th-TH', {
  dateStyle: 'full'
});

// รูปแบบที่ 6: เวลาแบบกลาง
const convertDate16 = now.toLocaleTimeString('th-TH', {
  timeStyle: 'medium'
});

// รูปแบบที่ 7: วันที่และเวลาแบบสั้น
const convertDate17 = now.toLocaleString('th-TH', {
  dateStyle: 'short',
  timeStyle: 'short'
});

// รูปแบบที่ 8: วันที่แบบสั้นและเวลาแบบเต็ม
var convertDate18 = new Date().toLocaleDateString("th-TH", {
  day: "numeric",
  month: "short",
  year: "numeric",
  weekday: "short"
});

var convertTime18 = new Date().toLocaleTimeString("th-TH", {
  hour: "2-digit",
  minute: "2-digit",
  second: "2-digit",
  hour12: false
}) + " น.";

var fullDateTime1 = `${convertDate18} ${convertTime18}`;

// รูปแบบที่ 9: วันที่แบบสั้นและเวลาแบบสั้น
var convertDate19 = new Date().toLocaleDateString("th-TH", {
  day: "numeric",
  month: "short",
  year: "numeric",
});

var convertTime19 = new Date().toLocaleTimeString("th-TH", {
  hour: "2-digit",
  minute: "2-digit",
  hour12: false
}) + " น.";

var fullDateTime2 = `${convertDate19} ${convertTime19}`;
